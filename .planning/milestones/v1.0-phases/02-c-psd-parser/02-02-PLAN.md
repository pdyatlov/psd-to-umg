UENUM()

USTRUCT()
struct PSD2UMG_API FPsdTextRun
{
    GENERATED_BODY()
    FString  Content;
    FString  FontName;
    float    SizePx = 0.f;
    FLinearColor Color = FLinearColor::White;
    TEnumAsByte<ETextJustify::Type> Alignment = ETextJustify::Left;
};

USTRUCT()
struct PSD2UMG_API FPsdLayer
{
    GENERATED_BODY()
    FString          Name;
    EPsdLayerType    Type = EPsdLayerType::Unknown;
    FIntRect         Bounds;
    float            Opacity = 1.f;
    bool             bVisible = true;
    TArray<FPsdLayer> Children;
    TArray<uint8>    RGBAPixels;
    int32            PixelWidth = 0;
    int32            PixelHeight = 0;
    FPsdTextRun      Text;
};

USTRUCT()
struct PSD2UMG_API FPsdDocument
{
    GENERATED_BODY()
    FString    SourcePath;
    FIntPoint  CanvasSize = FIntPoint::ZeroValue;
    TArray<FPsdLayer> RootLayers;
};

// PsdDiagnostics.h
enum class EPsdDiagnosticSeverity : uint8 { Info, Warning, Error };
struct PSD2UMG_API FPsdDiagnostic
{
    EPsdDiagnosticSeverity Severity = EPsdDiagnosticSeverity::Info;
    FString LayerName;
    FString Message;
};
struct PSD2UMG_API FPsdParseDiagnostics
{
    TArray<FPsdDiagnostic> Entries;
    bool HasErrors() const;
    void AddInfo(const FString& LayerName, const FString& Message);
    void AddWarning(const FString& LayerName, const FString& Message);
    void AddError(const FString& LayerName, const FString& Message);
};

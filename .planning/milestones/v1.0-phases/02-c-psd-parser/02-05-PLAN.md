    2. Create `Source/PSD2UMG/Tests/PsdParserSpec.cpp`:
       ```cpp
       #if WITH_DEV_AUTOMATION_TESTS
       #include "Misc/AutomationTest.h"
       #include "Interfaces/IPluginManager.h"
       #include "Misc/Paths.h"
       #include "Parser/PsdParser.h"
       #include "Parser/PsdTypes.h"
       #include "Parser/PsdDiagnostics.h"

       BEGIN_DEFINE_SPEC(FPsdParserSpec, "PSD2UMG.Parser.MultiLayer",
           EAutomationTestFlags::EditorContext | EAutomationTestFlags::EngineFilter)
           FString FixturePath;
           FPsdDocument Doc;
           FPsdParseDiagnostics Diag;
           bool bParsed = false;
       END_DEFINE_SPEC(FPsdParserSpec)

       void FPsdParserSpec::Define()
       {
           BeforeEach([this]() {
               FixturePath = FPaths::Combine(
                   IPluginManager::Get().FindPlugin(TEXT("PSD2UMG"))->GetBaseDir(),
                   TEXT("Source/PSD2UMG/Tests/Fixtures/MultiLayer.psd"));
               bParsed = PSD2UMG::Parser::ParseFile(FixturePath, Doc, Diag);
           });

           Describe("ParseFile on MultiLayer.psd", [this]() {
               It("returns true with no errors", [this]() {
                   TestTrue("bParsed", bParsed);
                   TestFalse("HasErrors", Diag.HasErrors());
               });
               It("produces correct canvas size", [this]() {
                   TestEqual("CanvasSize", Doc.CanvasSize, FIntPoint(256, 128));
               });
               It("has 3 root layers", [this]() {
                   TestEqual("RootLayers.Num", Doc.RootLayers.Num(), 3);
               });
               // ... assertions for Title, Buttons group, BtnNormal, BtnHover, Background
               // (full list per <behavior> above)
           });
       }
       #endif
       ```

    3. Implement every assertion listed in `<behavior>`. Use `TestEqual`, `TestTrue`, `TestFalse`, and `IsNearlyEqual` for floats. For colors use a tolerance of 0.05 on each channel.

    4. Layer lookup helpers: write a small `FindLayerByName(const TArray<FPsdLayer>& Layers, const FString& Name)` lambda inside the spec.

    Per CONTEXT.md D-Test: Editor-only, runs via Session Frontend → Automation. No external test runner.
  </action>

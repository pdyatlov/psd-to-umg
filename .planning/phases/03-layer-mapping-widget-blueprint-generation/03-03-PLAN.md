```cpp
#pragma once
#include "CoreMinimal.h"
#include "Components/CanvasPanelSlot.h"

struct FAnchorResult
{
    FAnchors Anchors;
    bool bStretchH = false;
    bool bStretchV = false;
    FString CleanName; // Layer name with suffix stripped
};

class FAnchorCalculator
{
public:
    // Returns anchor + whether stretch is active on each axis + cleaned name
    static FAnchorResult Calculate(const FString& LayerName, const FIntRect& Bounds, const FIntPoint& CanvasSize);

private:
    static bool TryParseSuffix(const FString& Name, FAnchors& OutAnchors, bool& bOutStretchH, bool& bOutStretchV, FString& OutCleanName);
    static FAnchors QuadrantAnchor(const FIntRect& Bounds, const FIntPoint& CanvasSize, bool& bOutStretchH, bool& bOutStretchV);
};
```

**FAnchorCalculator.cpp**:

`TryParseSuffix`: Check layer name for these suffixes (check longest first to avoid partial matches). If found, strip suffix from name and set anchor:

| Suffix | Anchors | StretchH | StretchV |
|--------|---------|----------|----------|
| `_anchor-tl` | (0,0) | false | false |
| `_anchor-tc` | (0.5,0) | false | false |
| `_anchor-tr` | (1,0) | false | false |
| `_anchor-cl` | (0,0.5) | false | false |
| `_anchor-c` | (0.5,0.5) | false | false |
| `_anchor-cr` | (1,0.5) | false | false |
| `_anchor-bl` | (0,1) | false | false |
| `_anchor-bc` | (0.5,1) | false | false |
| `_anchor-br` | (1,1) | false | false |
| `_stretch-h` | computed | true | false |
| `_stretch-v` | computed | false | true |
| `_fill` | (0,0,1,1) | true | true |

For `_stretch-h`: Y zone from vertical quadrant, anchors = (0, Y, 1, Y).
For `_stretch-v`: X zone from horizontal quadrant, anchors = (X, 0, X, 1).
Use `Layer.Name.EndsWith()` for suffix detection. Strip suffix via `Layer.Name.LeftChop(SuffixLen)`.

`QuadrantAnchor` (per D-05, D-06):
```
CX = (Bounds.Min.X + Bounds.Max.X) * 0.5f
CY = (Bounds.Min.Y + Bounds.Max.Y) * 0.5f
W = CanvasSize.X, H = CanvasSize.Y

bStretchH = Bounds.Width() >= W * 0.8f
bStretchV = Bounds.Height() >= H * 0.8f

if (bStretchH && bStretchV) return FAnchors(0,0,1,1)

VZone = CY < H/3 ? 0.f : CY < H*2/3 ? 0.5f : 1.f
HZone = CX < W/3 ? 0.f : CX < W*2/3 ? 0.5f : 1.f

if (bStretchH) return FAnchors(0, VZone, 1, VZone)
if (bStretchV) return FAnchors(HZone, 0, HZone, 1)
return FAnchors(HZone, VZone)  // point anchor
```

`Calculate`: Try suffix first. If no suffix matched, use QuadrantAnchor. Return FAnchorResult with CleanName (original name if no suffix).

```cpp
static void PopulateCanvas(FLayerMappingRegistry& Registry, UWidgetTree* Tree,
    UCanvasPanel* Parent, const TArray<FPsdLayer>& Layers, const FPsdDocument& Doc,
    const FIntPoint& CanvasSize)
{
    const int32 TotalLayers = Layers.Num();
    for (int32 i = 0; i < TotalLayers; ++i)
    {
        const FPsdLayer& Layer = Layers[i];

        // Skip invisible or zero-size per D-08, D-14
        if (!Layer.bVisible) { UE_LOG(LogPSD2UMG, Log, TEXT("Skipping invisible layer: %s"), *Layer.Name); continue; }
        if (Layer.Bounds.IsEmpty() && Layer.Type != EPsdLayerType::Group)
        {
            UE_LOG(LogPSD2UMG, Warning, TEXT("Skipping zero-size layer: %s"), *Layer.Name);
            continue;
        }

        UWidget* Widget = Registry.MapLayer(Layer, Doc, Tree);
        if (!Widget)
        {
            UE_LOG(LogPSD2UMG, Warning, TEXT("No mapper for layer: %s (type=%d)"), *Layer.Name, (int32)Layer.Type);
            continue; // D-08 skip+warn
        }

        // Add to canvas and configure slot
        UCanvasPanelSlot* Slot = Parent->AddChildToCanvas(Widget);
        if (Slot)
        {
            // Anchor calculation
            FAnchorResult AnchorResult = FAnchorCalculator::Calculate(Layer.Name, Layer.Bounds, CanvasSize);

            // Build FAnchorData
            FAnchorData Data;
            Data.Anchors = AnchorResult.Anchors;
            Data.Alignment = FVector2D(0.f, 0.f);

            // Coordinate math depends on stretch mode
            if (AnchorResult.bStretchH && AnchorResult.bStretchV)
            {
                // Full stretch: offsets are margins from all 4 edges
                Data.Offsets = FMargin(
                    Bounds.Min.X, Bounds.Min.Y,
                    CanvasSize.X - Bounds.Max.X,
                    CanvasSize.Y - Bounds.Max.Y);
            }
            else if (AnchorResult.bStretchH)
            {
                // Horizontal stretch: Left/Right are margins, Top is offset from anchor Y, Bottom is height
                float AnchorPixelY = AnchorResult.Anchors.Minimum.Y * CanvasSize.Y;
                Data.Offsets = FMargin(
                    Layer.Bounds.Min.X,
                    Layer.Bounds.Min.Y - AnchorPixelY,
                    CanvasSize.X - Layer.Bounds.Max.X,
                    Layer.Bounds.Height());
            }
            else if (AnchorResult.bStretchV)
            {
                // Vertical stretch: Top/Bottom are margins, Left is offset from anchor X, Right is width
                float AnchorPixelX = AnchorResult.Anchors.Minimum.X * CanvasSize.X;
                Data.Offsets = FMargin(
                    Layer.Bounds.Min.X - AnchorPixelX,
                    Layer.Bounds.Min.Y,
                    Layer.Bounds.Width(),
                    CanvasSize.Y - Layer.Bounds.Max.Y);
            }
            else
            {
                // Point anchor: Left=X offset from anchor, Top=Y offset from anchor, Right=Width, Bottom=Height
                float AnchorPixelX = AnchorResult.Anchors.Minimum.X * CanvasSize.X;
                float AnchorPixelY = AnchorResult.Anchors.Minimum.Y * CanvasSize.Y;
                Data.Offsets = FMargin(
                    Layer.Bounds.Min.X - AnchorPixelX,
                    Layer.Bounds.Min.Y - AnchorPixelY,
                    Layer.Bounds.Width(),
                    Layer.Bounds.Height());
            }

            Slot->SetLayout(Data);
            // ZOrder: PSD index 0 = topmost. UMG higher = on top. Invert.
            Slot->SetZOrder(TotalLayers - 1 - i);
        }

        // Recurse into group children
        if (Layer.Type == EPsdLayerType::Group && !Layer.Children.IsEmpty())
        {
            UCanvasPanel* ChildCanvas = Cast<UCanvasPanel>(Widget);
            if (ChildCanvas)
            {
                PopulateCanvas(Registry, Tree, ChildCanvas, Layer.Children, Doc, CanvasSize);
            }
        }
    }
}
```

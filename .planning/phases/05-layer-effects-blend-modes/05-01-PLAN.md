lrFX binary format (PSD spec):
```
uint16  version (= 0)
uint16  effectCount
For each effect:
  char[4] signature ("8BIM")
  char[4] effectKey:
    "dsdw" = drop shadow
    "isdw" = inner shadow
    "oglw" = outer glow
    "iglw" = inner glow
    "bevl" = bevel/emboss
    "sofi" = solid fill (color overlay)
  uint32  effectSize (bytes following this field)
  ... effect-specific payload
```

Drop shadow ("dsdw") payload (after size field):
```
uint32  version (= 0 or 2)
uint32  blur (pixels, fixed point 16.16 — divide by 65536)
uint32  intensity (fixed point)
uint32  angle (degrees, fixed point)
uint32  distance (pixels, fixed point 16.16)
uint16  colorSpace (= 0 for RGB)
uint16[4] color (R, G, B, unused — each 0..65535)
  NOTE: PSD uses native RGB here, NOT ARGB. Verify at runtime.
uint8   sign (= 0 for down/right positive)
... (opacity byte, enabled byte, etc.)
```

Color overlay / solid fill ("sofi") payload:
```
uint32  version (= 2)
uint32  key (blend mode)
uint16  colorSpace
uint16[4] color (R, G, B, unused — each 0..65535)
uint8   opacity (0-255)
uint8   enabled
```

In PsdParser.cpp, inside the `PSD2UMG::Parser::Internal` namespace, add a new static helper function `ExtractLayerEffects`:

```cpp
static void ExtractLayerEffects(
    const std::shared_ptr<PsdLayer>& InLayer,
    FPsdLayer& OutLayer,
    FPsdParseDiagnostics& OutDiag)
{
    // Scan unparsed_blocks() for the lrFX tagged block
    for (const auto& Block : InLayer->unparsed_blocks())
    {
        if (!Block || Block->getKey() != NAMESPACE_PSAPI::Enum::TaggedBlockKey::fxLayer)
            continue;

        const auto& Data = Block->m_Data;
        if (Data.size() < 4) break; // need at least version + count

        // Parse lrFX header
        size_t Pos = 0;
        auto ReadU16 = [&]() -> uint16 {
            if (Pos + 2 > Data.size()) return 0;
            uint16 Val = (static_cast<uint8>(Data[Pos]) << 8) | static_cast<uint8>(Data[Pos + 1]);
            Pos += 2;
            return Val;
        };
        auto ReadU32 = [&]() -> uint32 {
            if (Pos + 4 > Data.size()) return 0;
            uint32 Val = (static_cast<uint8>(Data[Pos]) << 24)
                       | (static_cast<uint8>(Data[Pos + 1]) << 16)
                       | (static_cast<uint8>(Data[Pos + 2]) << 8)
                       | static_cast<uint8>(Data[Pos + 3]);
            Pos += 4;
            return Val;
        };
        auto ReadU8 = [&]() -> uint8 {
            if (Pos + 1 > Data.size()) return 0;
            return static_cast<uint8>(Data[Pos++]);
        };

        uint16 Version = ReadU16();
        uint16 EffectCount = ReadU16();

        for (uint16 e = 0; e < EffectCount; ++e)
        {
            if (Pos + 8 > Data.size()) break;

            // Read signature (4 bytes, expect "8BIM")
            Pos += 4; // skip signature

            // Read effect key (4 bytes)
            char Key[5] = {};
            for (int k = 0; k < 4; ++k)
                Key[k] = static_cast<char>(Data[Pos++]);

            uint32 EffectSize = ReadU32();
            size_t EffectEnd = Pos + EffectSize;
            if (EffectEnd > Data.size()) EffectEnd = Data.size();

            if (FCStringAnsi::Strcmp(Key, "sofi") == 0)
            {
                // Color overlay / solid fill
                OutLayer.Effects.bHasColorOverlay = true;
                // Parse: version(4) + blendmode_key(4) + colorspace(2) + color[4](2 each) + opacity(1) + enabled(1)
                uint32 SofiVer = ReadU32();
                uint32 BlendKey = ReadU32();
                uint16 ColorSpace = ReadU16();
                // Color components: R, G, B, unused — each uint16 (0-65535)
                float R = static_cast<float>(ReadU16()) / 65535.f;
                float G = static_cast<float>(ReadU16()) / 65535.f;
                float B = static_cast<float>(ReadU16()) / 65535.f;
                ReadU16(); // unused
                uint8 OpacityByte = ReadU8();
                uint8 Enabled = ReadU8();
                float A = static_cast<float>(OpacityByte) / 255.f;
                OutLayer.Effects.ColorOverlayColor = Enabled ? FLinearColor(R, G, B, A) : FLinearColor::White;
                OutLayer.Effects.bHasColorOverlay = (Enabled != 0);
            }
            else if (FCStringAnsi::Strcmp(Key, "dsdw") == 0)
            {
                // Drop shadow
                OutLayer.Effects.bHasDropShadow = true;
                // Parse: version(4) + blur(4) + intensity(4) + angle(4) + distance(4) + colorSpace(2) + color(8) ...
                uint32 DsdwVer = ReadU32();
                uint32 BlurFixed = ReadU32();
                uint32 IntensityFixed = ReadU32();
                uint32 AngleFixed = ReadU32();
                uint32 DistanceFixed = ReadU32();
                float DistancePx = static_cast<float>(DistanceFixed) / 65536.f;
                float AngleDeg = static_cast<float>(AngleFixed) / 65536.f;
                // Convert angle+distance to XY offset
                float AngleRad = FMath::DegreesToRadians(AngleDeg);
                float OffsetX = DistancePx * FMath::Cos(AngleRad);
                float OffsetY = DistancePx * FMath::Sin(AngleRad);
                OutLayer.Effects.DropShadowOffset = FVector2D(OffsetX, OffsetY);

                uint16 ColorSpace = ReadU16();
                float R = static_cast<float>(ReadU16()) / 65535.f;
                float G = static_cast<float>(ReadU16()) / 65535.f;
                float B = static_cast<float>(ReadU16()) / 65535.f;
                ReadU16(); // unused
                // After color: sign(1), opacity byte may follow — try to read
                // The exact layout varies by version. Read opacity if available.
                uint8 Sign = ReadU8();
                // Negate Y if sign indicates upward
                if (Sign != 0) OffsetY = -OffsetY;
                OutLayer.Effects.DropShadowOffset = FVector2D(OffsetX, OffsetY);

                // Try to read opacity (next byte after sign in v0)
                uint8 ShadowOpacity = (Pos < EffectEnd) ? ReadU8() : 128;
                uint8 ShadowEnabled = (Pos < EffectEnd) ? ReadU8() : 1;
                float A = static_cast<float>(ShadowOpacity) / 255.f;
                OutLayer.Effects.DropShadowColor = ShadowEnabled ? FLinearColor(R, G, B, A) : FLinearColor(0, 0, 0, 0);
                OutLayer.Effects.bHasDropShadow = (ShadowEnabled != 0);
            }
            else if (FCStringAnsi::Strcmp(Key, "isdw") == 0
                  || FCStringAnsi::Strcmp(Key, "bevl") == 0)
            {
                // Inner shadow or bevel/emboss — complex effects (per D-09)
                OutLayer.Effects.bHasComplexEffects = true;
            }
            // Note: gradient overlay and pattern overlay use different block types
            // (lfx2 descriptor format, not lrFX). If they appear in lrFX, they would
            // have unknown keys — ignored gracefully.

            Pos = EffectEnd; // advance to next effect
        }
        break; // only one lrFX block expected
    }

    // Log extraction results
    if (OutLayer.Effects.bHasColorOverlay || OutLayer.Effects.bHasDropShadow || OutLayer.Effects.bHasComplexEffects)
    {
        UE_LOG(LogPSD2UMG, Log, TEXT("Layer '%s' effects: ColorOverlay=%d DropShadow=%d Complex=%d"),
            *OutLayer.Name,
            OutLayer.Effects.bHasColorOverlay,
            OutLayer.Effects.bHasDropShadow,
            OutLayer.Effects.bHasComplexEffects);
    }
}
```
